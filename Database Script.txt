-- Create the "weather_api" database (if it doesn't exist)
CREATE DATABASE weather_api;

-- Use the "weather_api" database
USE weather_api;

-- Create the "weather_forecast" table with an auto-incrementing "id" column as the primary key
CREATE TABLE weather_forecast (
    id INT AUTO_INCREMENT PRIMARY KEY,
    dt DATETIME,
    temperature DECIMAL(10, 2),
    humidity INT,
    weather_description VARCHAR(255),
    wind_speed DECIMAL(10, 2),
    wind_direction INT,
    clouds INT
);
